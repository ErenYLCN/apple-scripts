on run argv	tell application "iTerm 2"		tell current session of current window to write text "cd ~/" & (item 1 of argv) & " && nvim"				set utilWindow to create window with default profile				tell current session of utilWindow to write text "cd ~/" & (item 1 of argv)				tell utilWindow to set serverTab to create tab with default profile				tell current session of serverTab to write text "cd ~/" & (item 1 of argv) & " && npm run dev"	end tell		tell application "Google Chrome"		set activeIndex to get active tab index of window 1		tell window 1			set newTab to make new tab with properties {URL:"localhost:3000"}		end tell		set active tab index of window 1 to newTab	end tellend run
